3   public class Room {
4   
5       private int left;
6       private int top;
7       private int width;
8       private int height;
9   
10      public Room(int left, int top, int width, int height) {
11          this.left = left;
12          this.top = top;
13          this.width = width;
14          this.height = height;
15      }
16  
17      public int[][] getRoom() {
18          int[][] p = new int[this.height][this.width];
19          for (int y = 0; y < this.height; y++) {
20              for (int x = 0; x < this.width; x++) {
21                  p[y][x] = 1;
22              }
23          }
24          return p;
25      }
26  
27      public int getTop() {
28          return this.top;
29      }
30  
31      public int getLeft() {
32          return this.left;
33      }
34  
35      public int getRight() {
36          return left + width - 1;
37      }
38  
39      public int getBottom() {
40          return top + height - 1;
41      }
42  
43      public int getCenterX() {
44          return left + width / 2;
45      }
46  
47      public int getCenterY() {
48          return top + height / 2;
49      }
50  
51      public int getHeight() {
52          return this.height;
53      }
54  
55      public int getWidth() {
56          return this.width;
57      }
58  
59      public boolean collidesWith(Room other) {
60          if (left > other.getRight() - 1
61                  || top > other.getBottom() - 1
62                  || getRight() < other.left + 1
63                  || getBottom() < other.top + 1) {
64              return false;
65          }
66          return true;
67      }
68  }
