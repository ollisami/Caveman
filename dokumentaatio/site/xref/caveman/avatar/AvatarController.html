16      private GameController gameController;
17      private Avatar player;
18      private List<Avatar> enemies;
19      private int enemyCount = 4;
20  
21      public AvatarController(GameController gameController) {
22          this.gameController = gameController;
23          this.enemies = new ArrayList<>();
24      }
25  
26      /**
27       * Luo pelaajan, viholliset ym. Avatarit tyhjiin huoneisiin
28       *
29       */
30      public void spawnCharacters() {
31          Map map = gameController.getCurrentMap();
32          Room r = map.getEmptyRoom();
33          if (r == null) {
34              System.out.println("ERR! Empty room!");
35              return;
36          }
37          this.player = new Player(3, r.getCenterY(), r.getCenterX(), 100);
38          map.setData(player.getPosY(), player.getPosX(), player.getSpriteValue());
39  
40          while (enemies.size() < enemyCount) {
41              Room room = map.getEmptyRoom();
42              if (room == null) {
43                  System.out.println("Could not found room for enemy :(");
44                  break;
45              }
46              Enemy e = new Enemy(4, room.getCenterY(), room.getCenterX(), 100);
47              enemies.add(e);
48              map.setData(e.getPosY(), e.getPosX(), e.getSpriteValue());
49          }
50      }
51  
52      /**
53       * Liikuttaa avatareja.Aluksi liikutetaan pelaajaa parametrina annettuun
54       * suuntaan, jonka j√§lkeen liikutetaan viholliset kohti pelaajaa.
55       *
56       * @param py pelaajan suunta y
57       * @param px pelaajan suunta x
58       */
59      public void moveAvatars(int py, int px) {
60          Map map = gameController.getCurrentMap();
61          if (map.getData(this.player.getPosY() + py, this.player.getPosX() + px) == 1) {
62              map.setData(player.getPosY(), player.getPosX(), 1);
63              this.player.move(py, px);
64              map.setData(player.getPosY(), player.getPosX(), player.getSpriteValue());
65          }
66          py = this.player.getPosY();
67          px = this.player.getPosX();
68          for (Avatar e : this.enemies) {
69              int x = 0;
70              int y = 0;
71              if (e.getPosY() != py) {
72                  if (e.getPosY() < py) {
73                      y = 1;
74                  } else if (e.getPosY() > py) {
75                      y = -1;
76                  }
77                  if (map.getData(e.getPosY() + y, e.getPosX() + x) != 2) {
78                      map.setData(e.getPosY(), e.getPosX(), 1);
79                      e.move(y, x);
80                      map.setData(e.getPosY(), e.getPosX(), e.getSpriteValue());
81                      continue;
82                  }
83              }
84              y = 0;
85              if (e.getPosX() < px) {
86                  x = 1;
87              } else if (e.getPosX() > px) {
88                  x = -1;
89              }
90              if (map.getData(e.getPosY() + y, e.getPosX() + x) != 2) {
91                  map.setData(e.getPosY(), e.getPosX(), 1);
92                  e.move(y, x);
93                  map.setData(e.getPosY(), e.getPosX(), e.getSpriteValue());
94              }
95          }
96      }
